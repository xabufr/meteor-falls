00303    {
00304       clear(); // Delete mesh data

00364 {
00365    if (mRenderMethod == BB_METHOD_ACCELERATED)
00366    {
00367       //Delete old entity and mesh data
00368       if (mpEntity)
00369       {
00370          //Delete entity
00371          mpSceneNode->detachAllObjects();
00372          mpEntity->_getManager()->destroyEntity(mpEntity);
00373          mpEntity = NULL;
00374 
00375          //Delete mesh
00376          assert(!mPtrMesh.isNull());
00377          String meshName(mPtrMesh->getName());
00378          mPtrMesh.setNull();
00379          MeshManager::getSingleton().remove(meshName);
00380       }
00381 
00382       //If there are no billboards to create, exit
00383       if (mBillboardBuffer.empty())
00384          return;
00385 
00386       //Create manual mesh to store billboard quads
00387       mPtrMesh = MeshManager::getSingleton().createManual(getUniqueID("SBSmesh"), ResourceGroupManager::DEFAULT_RESOURCE_GROUP_NAME);
00388       Ogre::SubMesh *pSubMesh = mPtrMesh->createSubMesh();
00389       pSubMesh->useSharedVertices = false;
00390 
00391       //Setup vertex format information
00392       pSubMesh->vertexData = new VertexData;
00393       pSubMesh->vertexData->vertexStart = 0;
00394       pSubMesh->vertexData->vertexCount = 4 * mBillboardBuffer.size();
00395 
00396       VertexDeclaration* dcl = pSubMesh->vertexData->vertexDeclaration;
00397       size_t offset = 0;
00398       dcl->addElement(0, offset, VET_FLOAT3, VES_POSITION);
00399       offset += VertexElement::getTypeSize(VET_FLOAT3);
00400       dcl->addElement(0, offset, VET_FLOAT3, VES_NORMAL);
00401       offset += VertexElement::getTypeSize(VET_FLOAT3);
00402       dcl->addElement(0, offset, VET_COLOUR, VES_DIFFUSE);
00403       offset += VertexElement::getTypeSize(VET_COLOUR);
00404       dcl->addElement(0, offset, VET_FLOAT2, VES_TEXTURE_COORDINATES);
00405       offset += VertexElement::getTypeSize(VET_FLOAT2);
00406 
00407       //Populate a new vertex buffer
00408       HardwareVertexBufferSharedPtr vbuf = HardwareBufferManager::getSingleton().createVertexBuffer(
00409          offset, pSubMesh->vertexData->vertexCount, HardwareBuffer::HBU_STATIC_WRITE_ONLY, false);
00410       float* pReal = static_cast<float*>(vbuf->lock(HardwareBuffer::HBL_DISCARD));
00411 
00412       float minX = (float)Math::POS_INFINITY, minY = (float)Math::POS_INFINITY, minZ = (float)Math::POS_INFINITY;
00413       float maxX = (float)Math::NEG_INFINITY, maxY = (float)Math::NEG_INFINITY, maxZ = (float)Math::NEG_INFINITY;
00414 
00415       // For each billboard
00416       size_t billboardCount = mBillboardBuffer.size();
00417       for (size_t ibb = 0; ibb < billboardCount; ++ibb)
00418       {
00419          const StaticBillboard *bb = mBillboardBuffer[ibb];
00420 
00421          // position
00425 
00429          memcpy(pReal, &(bb->xPos), 5 * sizeof(float));
00430          pReal += 5; // move to next float value
00431          *pReal++ = 0.0f;
00432 
00433          // color
00434          *(reinterpret_cast< uint32* >(pReal++)) = bb->color;
00435          // uv
00436          *pReal++ = (bb->texcoordIndexU * mfUFactor);
00437          *pReal++ = (bb->texcoordIndexV * mfVFactor);
00438 
00439 
00440          // position
00447          memcpy(pReal, &(bb->xPos), 5 * sizeof(float));
00448          pReal += 5; // move to next float value
00449          *pReal++ = 1.0f;
00450          // color
00451          *(reinterpret_cast< uint32* >(pReal++)) = bb->color;
00452          // uv
00453          *pReal++ = ((bb->texcoordIndexU + 1) * mfUFactor);
00454          *pReal++ = (bb->texcoordIndexV * mfVFactor);
00455 
00456 
00457          // position
00464          memcpy(pReal, &(bb->xPos), 5 * sizeof(float));
00465          pReal += 5; // move to next float value
00466          *pReal++ = 2.0f;
00467          // color
00468          *(reinterpret_cast< uint32* > (pReal++)) = bb->color;
00469          // uv
00470          *pReal++ = (bb->texcoordIndexU * mfUFactor);
00471          *pReal++ = ((bb->texcoordIndexV + 1) * mfVFactor);
00472 
00473 
00474          // position
00481          memcpy(pReal, &(bb->xPos), 5 * sizeof(float));
00482          pReal += 5; // move to next float value
00483          *pReal++ = 3.0f;
00484          // color
00485          *(reinterpret_cast< uint32* >(pReal++)) = bb->color;
00486          // uv
00487          *pReal++ = ((bb->texcoordIndexU + 1) * mfUFactor);
00488          *pReal++ = ((bb->texcoordIndexV + 1) * mfVFactor);
00489 
00490          //Update bounding box
00491          if (bb->xPos - bb->xScaleHalf < minX) minX = bb->xPos - bb->xScaleHalf;
00492          if (bb->xPos + bb->xScaleHalf > maxX) maxX = bb->xPos + bb->xScaleHalf;
00493          if (bb->yPos - bb->yScaleHalf < minY) minY = bb->yPos - bb->yScaleHalf;
00494          if (bb->yPos + bb->yScaleHalf > maxY) maxY = bb->yPos + bb->yScaleHalf;
00495          //if (bb->zPos - halfXScale < minZ) minZ = bb->zPos - halfXScale;
00496          //if (bb->zPos + halfXScale > maxZ) maxZ = bb->zPos + halfXScale;
00497          if (bb->zPos < minZ) minZ = bb->zPos - 0.5f;
00498          if (bb->zPos > maxZ) maxZ = bb->zPos + 0.5f;
00499 
00500          delete bb;
00501       }
00502       mBillboardBuffer.clear(); //Empty the mBillboardBuffer now, because all billboards have been built
00503 
00504       vbuf->unlock();
00505       pSubMesh->vertexData->vertexBufferBinding->setBinding(0, vbuf);
00506 
00507       // Populate index buffer
00508       {
00509          pSubMesh->indexData->indexStart = 0;
00510          pSubMesh->indexData->indexCount = 6 * billboardCount;
00511          assert(pSubMesh->indexData->indexCount <= std::numeric_limits<Ogre::uint16>::max() && "To many indices. Use 32 bit indices");
00512          Ogre::HardwareIndexBufferSharedPtr &ptrIndBuf = pSubMesh->indexData->indexBuffer;
00513          ptrIndBuf = HardwareBufferManager::getSingleton().createIndexBuffer(HardwareIndexBuffer::IT_16BIT,
00514             pSubMesh->indexData->indexCount, HardwareBuffer::HBU_STATIC_WRITE_ONLY, false);
00515 
00516          Ogre::uint16 *pIBuf = static_cast < Ogre::uint16* > (ptrIndBuf->lock(HardwareBuffer::HBL_DISCARD));
00517          for (Ogre::uint16 i = 0; i < billboardCount; ++i)
00518          {
00519             Ogre::uint16 offset = i * 4;
00520 
00521             *pIBuf++ = 0 + offset;
00522             *pIBuf++ = 2 + offset;
00523             *pIBuf++ = 1 + offset;
00524 
00525             *pIBuf++ = 1 + offset;
00526             *pIBuf++ = 2 + offset;
00527             *pIBuf++ = 3 + offset;
00528          }
00529          ptrIndBuf->unlock(); // Unlock buffer and update GPU
00530       }
00531 
00532       // Finish up mesh
00533       {
00534          AxisAlignedBox bounds(minX, minY, minZ, maxX, maxY, maxZ);
00535          mPtrMesh->_setBounds(bounds);
00536          Vector3 temp = bounds.getMaximum() - bounds.getMinimum();
00537          mPtrMesh->_setBoundingSphereRadius(temp.length() * 0.5f);
00538 
00539          // Loading mesh
00540          Ogre::LoggingLevel logLev = LogManager::getSingleton().getDefaultLog()->getLogDetail();
00541          LogManager::getSingleton().setLogDetail(LL_LOW);
00542          mPtrMesh->load();
00543          LogManager::getSingleton().setLogDetail(logLev);
00544       }
00545 
00546       // Create an entity for the mesh
00547       mpEntity = mpSceneMgr->createEntity(mEntityName, mPtrMesh->getName(), mPtrMesh->getGroup());
00548       mpEntity->setCastShadows(false);
00549       mpEntity->setMaterial(mFadeEnabled ? mPtrFadeMaterial : mPtrMaterial);
00550 
00551       // Add to scene
00552       mpSceneNode->attachObject(mpEntity);
00553       mpEntity->setVisible(mVisible);
00554    }
00555 }
