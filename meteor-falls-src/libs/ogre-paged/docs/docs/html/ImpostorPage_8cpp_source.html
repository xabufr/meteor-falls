00197 //-----------------------------------------------------------------------------
00199 void ImpostorPage::update()
00200 {
00201    if (m_mapImpostorBatches.empty())  // SVA speed up
00202       return;
00203 
00204     //Calculate the direction the impostor batches should be facing
00205     Vector3 camPos = m_pPagedGeom->_convertToLocal(m_pPagedGeom->getCamera()->getDerivedPosition());
00206     
00207     // Update all batches
00208    Ogre::Real distX = camPos.x - m_vecCenter.x;
00209     Ogre::Real distZ = camPos.z - m_vecCenter.z;
00210     Ogre::Real distY = camPos.y - m_vecCenter.y;
00211     Ogre::Real distRelZ = Math::Sqrt(distX * distX + distZ * distZ);
00212     Radian pitch = Math::ATan2(distY, distRelZ);
00213 
00214     Radian yaw;
00215     if (distRelZ > m_pPagedGeom->getPageSize() * 3)
00216    {
00217         yaw = Math::ATan2(distX, distZ);
00218     }
00219    else
00220    {
00221         Vector3 dir = m_pPagedGeom->_convertToLocal(m_pPagedGeom->getCamera()->getDerivedDirection());
00222         yaw = Math::ATan2(-dir.x, -dir.z);
00223     }
00224 
00225    TImpostorBatchs::iterator iter = m_mapImpostorBatches.begin(), iend = m_mapImpostorBatches.end();
00226    while (iter != iend)
00227    {
00228       iter->second->setAngle(pitch.valueDegrees(), yaw.valueDegrees());
00229       ++iter;
00230    }
00231 }

00297 //Rotates all the impostors to the specified angle (virtually - it actually changes
00298 //their materials to produce this same effect)
00299 void ImpostorBatch::setAngle(Ogre::Real pitchDeg, Ogre::Real yawDeg)
00300 {
00301     // Calculate pitch material index
00302     int newPitchIndex = 0;
00303 #ifdef IMPOSTOR_RENDER_ABOVE_ONLY
00304     if (pitchDeg > 0)
00305    {
00306         float maxPitchIndexDeg = (90.0f * (IMPOSTOR_PITCH_ANGLES-1)) / IMPOSTOR_PITCH_ANGLES;
00307         newPitchIndex = (int)(IMPOSTOR_PITCH_ANGLES * (pitchDeg / maxPitchIndexDeg));
00308         if (newPitchIndex > IMPOSTOR_PITCH_ANGLES-1) newPitchIndex = IMPOSTOR_PITCH_ANGLES-1;
00309     }
00310 #else
00311     float minPitchIndexDeg = -90.0f;
00312     float maxPitchIndexDeg = ((180.0f * (IMPOSTOR_PITCH_ANGLES-1)) / IMPOSTOR_PITCH_ANGLES) - 90.0f;
00313     newPitchIndex = (int)(IMPOSTOR_PITCH_ANGLES * ((pitchDeg - minPitchIndexDeg) / (maxPitchIndexDeg - minPitchIndexDeg)));
00314     if (newPitchIndex > IMPOSTOR_PITCH_ANGLES-1) newPitchIndex = IMPOSTOR_PITCH_ANGLES-1;
00315     if (newPitchIndex < 0) newPitchIndex = 0;
00316 #endif
00317     
00318     // Calculate yaw material index
00319    int newYawIndex = yawDeg > 0 ? int(IMPOSTOR_YAW_ANGLES * (yawDeg / 360.0f) + 0.5f) % IMPOSTOR_YAW_ANGLES :
00320       int(IMPOSTOR_YAW_ANGLES + IMPOSTOR_YAW_ANGLES * (yawDeg / 360.0f) + 0.5f) % IMPOSTOR_YAW_ANGLES;
00321     
00322     // Change materials if necessary
00323     if (newPitchIndex != pitchIndex || newYawIndex != yawIndex)
00324    {
00325         pitchIndex = newPitchIndex;
00326         yawIndex = newYawIndex;
00327         bbset->setMaterial(m_pTexture->material[pitchIndex][yawIndex]->getName());
00328     }
00329 }
