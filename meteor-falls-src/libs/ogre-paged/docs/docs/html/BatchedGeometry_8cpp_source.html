00106 void BatchedGeometry::addEntity(Entity *ent, const Vector3 &position,
00107                                 const Quaternion &orientation,
00108                                 const Vector3 &scale,
00109                                 const Ogre::ColourValue &color)
00110 {
00111    const MeshPtr &mesh = ent->getMesh();
00112 
00113    //If shared vertex data is used, extract into non-shared data
00114    extractVertexDataFromShared(mesh);   
00115 
00116    //For each subentity
00117    for (uint32 i = 0; i < ent->getNumSubEntities(); ++i)
00118    {
00119       //Get the subentity
00120       SubEntity *subEntity = ent->getSubEntity(i);
00121       SubMesh *subMesh = subEntity->getSubMesh();
00122 
00123       //Generate a format string that uniquely identifies this material & vertex/index format
00124       if (subMesh->vertexData == NULL)
00125          OGRE_EXCEPT(Exception::ERR_INTERNAL_ERROR, "SubMesh vertex data not found!", "BatchedGeometry::addEntity()");
00126       String formatStr = getFormatString(subEntity);
00127 
00128       //If a batch using an identical format exists...
00129       SubBatch *batch = 0;
00130       TSubBatchMap::iterator batchIter = m_mapSubBatch.find(formatStr);
00131       if (batchIter != m_mapSubBatch.end())
00132          batch = batchIter->second; //Use the batch
00133       else  // Otherwise create a new batch
00134       {
00135          batch = new SubBatch(this, subEntity);
00136          m_mapSubBatch.insert(std::pair<String, SubBatch*>(formatStr, batch));
00137       }
00138 
00139       //Now add the submesh to the compatible batch
00140       batch->addSubEntity(subEntity, position, orientation, scale, color);
00141    }
00142 
00143    //Update bounding box
00144    Matrix4 mat(orientation);
00145    mat.setScale(scale);
00146    AxisAlignedBox entBounds = ent->getBoundingBox();
00147    entBounds.transform(mat);
00148 
00149    if (m_BoundsUndefined)
00150    {
00151       m_boundsAAB.setMinimum(entBounds.getMinimum() + position);
00152       m_boundsAAB.setMaximum(entBounds.getMaximum() + position);
00153       m_BoundsUndefined = false;
00154    }
00155    else
00156    {
00157       Vector3 min = m_boundsAAB.getMinimum();
00158       Vector3 max = m_boundsAAB.getMaximum();
00159       min.makeFloor(entBounds.getMinimum() + position);
00160       max.makeCeil(entBounds.getMaximum() + position);
00161       m_boundsAAB.setMinimum(min);
00162       m_boundsAAB.setMaximum(max);
00163    }

00169 uint32 CountUsedVertices(IndexData *id, std::map<uint32, uint32> &ibmap)
00170 {
00171    uint32 i, count;
00172    switch (id->indexBuffer->getType()) {
00173       case HardwareIndexBuffer::IT_16BIT:
00174          {
00175             uint16 *data = (uint16*)id->indexBuffer->lock(id->indexStart * sizeof(uint16), 
00176                id->indexCount * sizeof(uint16), HardwareBuffer::HBL_READ_ONLY);
00177 
00178             for (i = 0; i < id->indexCount; i++) {
00179                uint16 index = data[i];
00180                if (ibmap.find(index) == ibmap.end()) ibmap[index] = (uint32)(ibmap.size());
00181             }
00182             count = (uint32)ibmap.size();
00183             id->indexBuffer->unlock();
00184          }
00185          break;
00186 
00187       case HardwareIndexBuffer::IT_32BIT:
00188          {
00189             uint32 *data = (uint32*)id->indexBuffer->lock(id->indexStart * sizeof(uint32), 
00190                id->indexCount * sizeof(uint32), HardwareBuffer::HBL_READ_ONLY);
00191 
00192             for (i = 0; i < id->indexCount; i++) {
00193                uint32 index = data[i];
00194                if (ibmap.find(index) == ibmap.end()) ibmap[index] = (uint32)(ibmap.size());
00195             }
00196             count = (uint32)ibmap.size();
00197             id->indexBuffer->unlock();
00198          }
00199          break;
00200 
00201       default:
00202          throw new Ogre::Exception(0, "Unknown index buffer type", "Converter.cpp::CountVertices");
00203          break;
00204    }

00212 void BatchedGeometry::extractVertexDataFromShared(const Ogre::MeshPtr &mesh)
00213 {
00214    if (mesh.isNull() || !mesh->sharedVertexData)
00215       return;
00216 
00217    Mesh::SubMeshIterator subMeshIterator = mesh->getSubMeshIterator();
00218 
00219    // Get shared vertex data
00220    VertexData *oldVertexData = mesh->sharedVertexData;
00221 
00222    while (subMeshIterator.hasMoreElements()) {
00223       SubMesh *subMesh = subMeshIterator.getNext();
00224 
00225       // Get index data
00226       IndexData *indexData = subMesh->indexData;
00227       HardwareIndexBufferSharedPtr ib = indexData->indexBuffer;
00228 
00229       // Create new nonshared vertex data
00230       std::map<uint32, uint32> indicesMap;
00231       VertexData *newVertexData = new VertexData();
00232       newVertexData->vertexCount = CountUsedVertices(indexData, indicesMap);
00233       //delete newVertexData->vertexDeclaration;
00234       newVertexData->vertexDeclaration = oldVertexData->vertexDeclaration->clone();
00235 
00236       // Create new vertex buffers
00237       uint32 buffersCount = (uint32)oldVertexData->vertexBufferBinding->getBufferCount();
00238       for (uint32 bufferIndex = 0; bufferIndex < buffersCount; bufferIndex++) {
00239 
00240          // Lock shared vertex buffer
00241          HardwareVertexBufferSharedPtr oldVertexBuffer = oldVertexData->vertexBufferBinding->getBuffer(bufferIndex);
00242          size_t vertexSize = oldVertexBuffer->getVertexSize();
00243          uint8 *oldLock = (uint8*)oldVertexBuffer->lock(0, oldVertexData->vertexCount * vertexSize, HardwareBuffer::HBL_READ_ONLY);
00244 
00245          // Create and lock nonshared vertex buffer
00246          HardwareVertexBufferSharedPtr newVertexBuffer = HardwareBufferManager::getSingleton().createVertexBuffer(
00247             vertexSize, newVertexData->vertexCount, oldVertexBuffer->getUsage(), oldVertexBuffer->hasShadowBuffer());
00248          uint8 *newLock = (uint8*)newVertexBuffer->lock(0, newVertexData->vertexCount * vertexSize, HardwareBuffer::HBL_NORMAL);
00249 
00250          // Copy vertices from shared vertex buffer into nonshared vertex buffer
00251          std::map<uint32, uint32>::iterator i, iend = indicesMap.end();
00252          for (i = indicesMap.begin(); i != iend; i++) {
00253             memcpy(newLock + vertexSize * i->second, oldLock + vertexSize * i->first, vertexSize);
00254          }
00255 
00256          // Unlock vertex buffers
00257          oldVertexBuffer->unlock();
00258          newVertexBuffer->unlock();
00259 
00260          // Bind new vertex buffer
00261          newVertexData->vertexBufferBinding->setBinding(bufferIndex, newVertexBuffer);
00262       }
00263 
00264       // Re-create index buffer
00265       switch (indexData->indexBuffer->getType()) {
00266          case HardwareIndexBuffer::IT_16BIT:
00267             {
00268                uint16 *data = (uint16*)indexData->indexBuffer->lock(indexData->indexStart * sizeof(uint16), 
00269                   indexData->indexCount * sizeof(uint16), HardwareBuffer::HBL_NORMAL);
00270 
00271                for (uint32 i = 0; i < indexData->indexCount; i++) {
00272                   data[i] = (uint16)indicesMap[data[i]];
00273                }
00274 
00275                indexData->indexBuffer->unlock();
00276             }
00277             break;
00278 
00279          case HardwareIndexBuffer::IT_32BIT:
00280             {
00281                uint32 *data = (uint32*)indexData->indexBuffer->lock(indexData->indexStart * sizeof(uint32), 
00282                   indexData->indexCount * sizeof(uint32), HardwareBuffer::HBL_NORMAL);
00283 
00284                for (uint32 i = 0; i < indexData->indexCount; i++) {
00285                   data[i] = (uint32)indicesMap[data[i]];
00286                }
00287 
00288                indexData->indexBuffer->unlock();
00289             }
00290             break;
00291 
00292          default:
00293             throw new Ogre::Exception(0, "Unknown index buffer type", "Converter.cpp::CountVertices");
00294             break;
00295       }
00296 
00297       // Store new attributes
00298       subMesh->useSharedVertices = false;
00299       subMesh->vertexData = newVertexData;
00300    }
00301 
00302    // Release shared vertex data
00303    delete mesh->sharedVertexData;
00304    mesh->sharedVertexData = NULL;
00305 }

00310 String BatchedGeometry::getFormatString(SubEntity *ent)
00311 {
00312    static char buf[1024];
00313    // add materialname and buffer type
00314    int countWritten =  sprintf(buf, "%s|%d", ent->getMaterialName().c_str(), ent->getSubMesh()->indexData->indexBuffer->getType());
00315 
00316    // now add vertex decl
00317    const VertexDeclaration::VertexElementList &elemList = ent->getSubMesh()->vertexData->vertexDeclaration->getElements();
00318    for (VertexDeclaration::VertexElementList::const_iterator i = elemList.begin(), iend = elemList.end(); i != iend; ++i)
00319    {
00320       const VertexElement &el = *i;
00321       countWritten += sprintf(buf + countWritten, "|%d|%d|%d", el.getSource(), el.getSemantic(), el.getType());
00322    }
00323 
00324    return buf;

00333    if (m_Built)
00334       OGRE_EXCEPT(Exception::ERR_DUPLICATE_ITEM, "Invalid call to build() - geometry is already batched (call clear() first)", "BatchedGeometry::GeomBatch::build()");
00335

00337    {
00338       // Finish bounds information
00339       m_vecCenter = m_boundsAAB.getCenter();                            // Calculate bounds center
00340       m_boundsAAB.setMinimum(m_boundsAAB.getMinimum() - m_vecCenter);   // Center the bounding box
00341       m_boundsAAB.setMaximum(m_boundsAAB.getMaximum() - m_vecCenter);   // Center the bounding box
00342       m_fRadius = m_boundsAAB.getMaximum().length();                    // Calculate BB radius

00346 
00347       //Build each batch
00348       for (TSubBatchMap::iterator i = m_mapSubBatch.begin(), iend = m_mapSubBatch.end(); i != iend; ++i)
00349          i->second->build();
00350 
00351       m_pSceneNode->attachObject(this);   // Attach the batch to the scene node
00352 
00353       //Debug

00423 void BatchedGeometry::_notifyCurrentCamera(Camera *cam)
00424 {
00425    if (getRenderingDistance() == Ogre::Real(0.))
00426       m_bWithinFarDistance = true;
00427    else
00428    {
00429       //Calculate camera distance
00430       Vector3 camVec = _convertToLocal(cam->getDerivedPosition()) - m_vecCenter;
00431       Real centerDistanceSquared = camVec.squaredLength();
00432       m_fMinDistanceSquared = std::max(Ogre::Real(0.), centerDistanceSquared - (m_fRadius * m_fRadius));
00433       //Note: centerDistanceSquared measures the distance between the camera and the center of the GeomBatch,
00434       //while minDistanceSquared measures the closest distance between the camera and the closest edge of the
00435       //geometry's bounding sphere.
00436 
00437       //Determine whether the BatchedGeometry is within the far rendering distance
00438       m_bWithinFarDistance = m_fMinDistanceSquared <= Math::Sqr(getRenderingDistance());
00439    }
00440 }

00572       HardwareIndexBuffer::IT_32BIT : HardwareIndexBuffer::IT_16BIT;
00573 
00574    //Allocate the index buffer

00576       destIndexType, m_pIndexData->indexCount, HardwareBuffer::HBU_STATIC_WRITE_ONLY);

00585 
00586    //Allocate & lock the vertex buffers
00587    std::vector<uchar*> vertexBuffers;
00588    std::vector<VertexDeclaration::VertexElementList> vertexBufferElements;
00589

00591    VertexDeclaration *vertDecl = m_pVertexData->vertexDeclaration;
00592

00594    {

00596          vertDecl->getVertexSize(i), m_pVertexData->vertexCount, HardwareBuffer::HBU_STATIC_WRITE_ONLY);
00597       vertBinding->setBinding(i, buffer);
00598 
00599       vertexBuffers.push_back(static_cast<uchar*>(buffer->lock(HardwareBuffer::HBL_DISCARD)));
00600       vertexBufferElements.push_back(vertDecl->findElementsBySource(i));
00601    }
00602

00604    if (m_RequireVertexColors)

00607       {

00609          vertDecl->addElement(i, 0, VET_COLOUR, VES_DIFFUSE);
00610 
00611          HardwareVertexBufferSharedPtr buffer = HardwareBufferManager::getSingleton().createVertexBuffer(
00612             vertDecl->getVertexSize(i), m_pVertexData->vertexCount, HardwareBuffer::HBU_STATIC_WRITE_ONLY);
00613          vertBinding->setBinding(i, buffer);
00614 
00615          vertexBuffers.push_back(static_cast<uchar*>(buffer->lock(HardwareBuffer::HBL_DISCARD)));
00616          vertexBufferElements.push_back(vertDecl->findElementsBySource(i));
00617       }
00618

00621
