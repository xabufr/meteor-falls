00098 {
00099     if (cam == NULL){
00100         //Simply set camera to null
00101         sceneCam = NULL;
00102     } else {
00103         if (sceneMgr && cam->getSceneManager() != sceneMgr)
00104             OGRE_EXCEPT(Exception::ERR_INVALIDPARAMS, "The specified camera is from the wrong SceneManager", "PagedGeometry::setCamera()");
00105 
00106         if (cam == lastSceneCam){
00107             //If the cache values for this camera are preserved, use them
00108             std::swap(oldCamPos, lastOldCamPos);
00109             std::swap(sceneCam, lastSceneCam);
00110         } else {
00111             lastSceneCam = sceneCam;
00112             lastOldCamPos = oldCamPos;
00113             sceneCam = cam;
00114         }
00115         
00116         //If sceneMgre is NULL (this only occurs the first time a camera is set),
00117         //then set the scene manager (it won't change after this point).
00118         if (sceneMgr == NULL)
00119             sceneMgr = sceneCam->getSceneManager();

00121         //If rootNode is NULL (this also only occurs the first time a camera is set),
00122         //the create a scene node (it won't change after this point) for the coordinate

00124         if (rootNode == NULL){
00125             #ifdef PAGEDGEOMETRY_ALTERNATE_COORDSYSTEM
00126             rootNode = sceneMgr->getRootSceneNode()->createChildSceneNode();

00211 void PagedGeometry::update()
00212 {
00213     //If no camera has been set, then return without doing anything
00214     if (sceneCam == NULL)
00215         return;
00216 
00217     //Calculate time since last update
00218     unsigned long deltaTime, tmp;
00219     tmp = timer.getMilliseconds();
00220     deltaTime = tmp - lastTime;
00221     lastTime = tmp;
00222 
00223     //Get camera position and speed
00224     Vector3 camPos = _convertToLocal(sceneCam->getDerivedPosition());
00225     Vector3 camSpeed;   //Speed in units-per-millisecond
00226     if (deltaTime == 0){
00227         camSpeed.x = 0;
00228         camSpeed.y = 0;
00229         camSpeed.z = 0;
00230     } else {
00231         camSpeed = (camPos - oldCamPos) / deltaTime;
00232     }
00233     oldCamPos = camPos;
00234     
00235     if (pageLoader != 0){
00236         //Update the PageLoader
00237         pageLoader->frameUpdate();
00238 
00239         //Update all the page managers
00240         bool enableCache = true;
00241         std::list<GeometryPageManager *>::iterator it;
00242         GeometryPageManager *prevMgr = NULL;
00243         for (it = managerList.begin(); it != managerList.end(); ++it)
00244       {
00245             GeometryPageManager *mgr = *it;
00246             mgr->update(deltaTime, camPos, camSpeed, enableCache, prevMgr);
00247             prevMgr = mgr;
00248         }
00249     }
00250 
00251     //Update misc. subsystems
00252    StaticBillboardSet::updateAll(_convertToLocal(sceneCam->getDerivedDirection()));
00253 }

00438 void GeometryPageManager::update(unsigned long deltaTime, Vector3 &camPos, Vector3 &camSpeed, bool &enableCache, GeometryPageManager *prevManager)
00439 {
00440     //-- Cache new geometry pages --
00441 
00442     //Cache 1 page ahead of the view ranges
00443     const Real cacheDist = farTransDist + mainGeom->getPageSize();
00444     const Real cacheDistSq = cacheDist * cacheDist;
00445     
00446     //First calculate the general area where the pages will be processed
00447     // 0,0 is the left top corner of the bounding box
00448     int x1 = (int)Math::Floor(((camPos.x - cacheDist) - gridBounds.left) / mainGeom->getPageSize());
00449     int x2 = (int)Math::Floor(((camPos.x + cacheDist) - gridBounds.left) / mainGeom->getPageSize());
00450     int z1 = (int)Math::Floor(((camPos.z - cacheDist) - gridBounds.top) / mainGeom->getPageSize());
00451     int z2 = (int)Math::Floor(((camPos.z + cacheDist) - gridBounds.top) / mainGeom->getPageSize());
00452 
00453     if (scrollBuffer)
00454     {
00455         //Check if the page grid needs to be scrolled
00456         int shiftX = 0, shiftZ = 0;
00457         if (x1 < 0) shiftX = x1; else if (x2 >= geomGridX-1) shiftX = x2 - (geomGridX-1);
00458         if (z1 < 0) shiftZ = z1; else if (z2 >= geomGridZ-1) shiftZ = z2 - (geomGridZ-1);
00459         if (shiftX != 0 || shiftZ != 0)
00460         {
00461             //Scroll grid
00462             _scrollGridPages(shiftX, shiftZ);
00463 
00464             //Update grid bounds and processing area
00465             gridBounds.left += shiftX * mainGeom->getPageSize();
00466             gridBounds.right += shiftX * mainGeom->getPageSize();
00467             gridBounds.top += shiftZ * mainGeom->getPageSize();
00468             gridBounds.bottom += shiftZ * mainGeom->getPageSize();
00469             x1 -= shiftX;  x2 -= shiftX;
00470             z1 -= shiftZ;  z2 -= shiftZ;
00471         }
00472     }
00473     else
00474     {
00475         // make sure that values are inbounds
00476         if(x2 >= geomGridX)
00477             x2 = geomGridX - 1;
00478         if(z2 >= geomGridZ)
00479             z2 = geomGridZ - 1;
00480 
00481         if (x1 < 0)
00482             x1 = 0;
00483         if (z1 < 0)
00484             z1 = 0;
00485     }
00486     //Now, in that general area, find what pages are within the cacheDist radius
00487     //Pages within the cacheDist radius will be added to the pending block list
00488     //to be loaded later, and pages within farDist will be loaded immediately.
00489     for (int x = x1; x <= x2; ++x){
00490         for (int z = z1; z <= z2; ++z){
00491             GeometryPage *blk = _getGridPage(x, z);
00492             
00493             Real dx = camPos.x - blk->_centerPoint.x;
00494             Real dz = camPos.z - blk->_centerPoint.z;
00495             Real distSq = dx * dx + dz * dz;
00496             
00497             //If the page is in the cache radius...
00498             if (distSq <= cacheDistSq){
00499                 //If the block hasn't been loaded yet, it should be
00500                 if (blk->_loaded == false){
00501                     //Test if the block's distance is between nearDist and farDist
00502                     if (distSq >= nearDistSq && distSq < farTransDistSq){
00503                         //If so, load the geometry immediately
00504                         _loadPage(blk);
00505                         loadedList.push_back(blk);
00506                         blk->_iter = (--loadedList.end());
00507 
00508                         //And remove it from the pending list if necessary
00509                         if (blk->_pending){
00510                             pendingList.remove(blk);
00511                             blk->_pending = false;
00512                         }
00513                     } else {
00514                         //Otherwise, add it to the pending geometry list (if not already)
00515                         //Pages in then pending list will be loaded later (see below)
00516                         if (!blk->_pending){
00517                             pendingList.push_back(blk);
00518                             blk->_pending = true;
00519                         }
00520                     }
00521                 } else {
00522                     //Set the inactive time to 0 (since the page is active). This
00523                     //must be done in order to keep it from expiring (and deleted).
00524                     //This way, blocks not in the cache radius won't have their
00525                     //inactivity clock reset, and they will expire in a few seconds.
00526                     blk->_inactiveTime = 0;
00527                 }
00528             }
00529         }
00530     }
00531     
00532     
00533     //Calculate cache speeds based on camera speed. This is important to keep the cache
00534     //process running smooth, because if the cache can't keep up with the camera, the
00535     //immediately visible pages will be forced to load instantly, which can cause
00536     //noticeable and sudden stuttering. The cache system results in smoother performance
00537     //because it smooths the loading tasks out across multiple frames. For example,
00538     //instead of loading 10+ blocks every 2 seconds, the cache would load 1 block every
00539     //200 milliseconds.
00540     Real speed = Math::Sqrt(camSpeed.x * camSpeed.x + camSpeed.z * camSpeed.z);
00541     
00542     unsigned long cacheInterval = 0;
00543     if (speed == 0)
00544         cacheInterval = maxCacheInterval;
00545     else {
00546         cacheInterval = (unsigned long)((mainGeom->getPageSize() * 0.8f) / (speed * pendingList.size()));
00547         if (cacheInterval > maxCacheInterval)
00548             cacheInterval = maxCacheInterval;
00549     }
00550 
00551 
00552     TPGeometryPages::iterator i1, i2;
00553 
00554     //Now load a single geometry page periodically, based on the cacheInterval
00555     cacheTimer += deltaTime;
00556     if (cacheTimer >= cacheInterval && enableCache){
00557         //Find a block to be loaded from the pending list
00558         i1 = pendingList.begin();
00559         i2 = pendingList.end();
00560         while (i1 != i2)
00561         {
00562             GeometryPage *blk = *i1;
00563             
00564             //Remove it from the pending list
00565             i1 = pendingList.erase(i1);
00566             blk->_pending = false;
00567             
00568             //If it's within the geometry cache radius, load it and break out of the loop
00569             Real dx = camPos.x - blk->_centerPoint.x;
00570             Real dz = camPos.z - blk->_centerPoint.z;
00571             Real distSq = dx * dx + dz * dz;
00572             if (distSq <= cacheDistSq){
00573                 _loadPage(blk);
00574                 loadedList.push_back(blk);
00575                 blk->_iter = (--loadedList.end());
00576 
00577                 enableCache = false;
00578                 break;
00579             }
00580 
00581             //Otherwise this will keep looping until an unloaded page is found
00582         }
00583             
00584         //Reset the cache timer
00585         cacheTimer = 0;
00586     }
00587 
00588 
00589     //-- Update existing geometry and impostors --
00590 
00591     //Loop through each loaded geometry block
00592     i1 = loadedList.begin();
00593     i2 = loadedList.end();
00594 
00595     //Real fadeBeginDistSq = farDistSq - Math::Sqr(mainGeom->getPageSize() * 1.4142f);
00596     Real halfPageSize = mainGeom->getPageSize() * 0.5f;
00597     while (i1 != i2)
00598     {
00599         GeometryPage *blk = *i1;
00600         
00601         //If the geometry has expired...
00602         if (blk->_inactiveTime >= inactivePageLife){
00603             if (!blk->_keepLoaded) {
00604                 //Unload it
00605                 _unloadPage(blk);
00606                 i1 = loadedList.erase(i1);
00607             } else {
00608                 //This page needs to be kept loaded indefinitely, so don't unload it
00609                 blk->_inactiveTime = 0;
00610                 ++i1;
00611             }
00612         } else {
00613             //Update it's visibility/fade status based on it's distance from the camera
00614             bool visible = false;
00615             Real dx = camPos.x - blk->_centerPoint.x;
00616             Real dz = camPos.z - blk->_centerPoint.z;
00617             Real distSq = dx * dx + dz * dz;
00618 
00619             Real overlap = 0, tmp;
00620 
00621             tmp = blk->_trueBounds.getMaximum().x - halfPageSize;
00622             if (tmp > overlap) overlap = tmp;
00623             tmp = blk->_trueBounds.getMaximum().z - halfPageSize;
00624             if (tmp > overlap) overlap = tmp;
00625             tmp = blk->_trueBounds.getMinimum().x + halfPageSize;
00626             if (tmp > overlap) overlap = tmp;
00627             tmp = blk->_trueBounds.getMinimum().z + halfPageSize;
00628             if (tmp > overlap) overlap = tmp;
00629 
00630             Real pageLengthSq = Math::Sqr((mainGeom->getPageSize() + overlap) * 1.41421356f);
00631         
00632             if (distSq + pageLengthSq >= nearDistSq && distSq - pageLengthSq < farTransDistSq){
00633                 //Fade the page when transitioning
00634                 bool enable = false;
00635                 Real fadeNear = 0;
00636                 Real fadeFar = 0;
00637                 
00638                 if (fadeEnabled && distSq + pageLengthSq >= farDistSq){
00639                     //Fade in
00640                     visible = true;
00641                     enable = true;
00642                     fadeNear = farDist;
00643                     fadeFar = farTransDist;
00644                 }
00645                 else if (prevManager && prevManager->fadeEnabled && (distSq - pageLengthSq < prevManager->farTransDistSq)){
