00095 void BatchPage::addEntity(Entity *ent, const Vector3 &position, const Quaternion &rotation,
00096                           const Vector3 &scale, const Ogre::ColourValue &color)
00097 {
00098     const size_t numManLod = ent->getNumManualLodLevels();
00099 
00100 #ifdef _DEBUG
00101     //Warns if using LOD batch and entities does not have enough LOD support.
00102     if (m_nLODLevel > 0 && numManLod < m_nLODLevel)
00103    {
00104         Ogre::LogManager::getSingleton().logMessage("BatchPage::addEntity: " + ent->getName() +
00105          " entity has less than " + Ogre::StringConverter::toString(m_nLODLevel) +
00106          " manual lod level(s). Performance warning.");
00107    }
00108 #endif
00109 
00110     if (m_nLODLevel == 0 || numManLod == 0)
00111         m_pBatchGeom->addEntity(ent, position, rotation, scale, color);
00112     else
00113     {
00114         const size_t bestLod = numManLod < m_nLODLevel - 1 ? numManLod : m_nLODLevel - 1;
00115         Ogre::Entity * lod = ent->getManualLodLevel(bestLod);
00116         m_pBatchGeom->addEntity(lod, position, rotation, scale, color);
00117     }
00118 }

00129    {

00131         const MaterialPtr &ptrMat = subBatch->getMaterial();

00134         //This is done because the default shader applied by BatchPage
00135         //doesn't support specular, and fixed-function needs to look
00136         //the same as the shader (for computers with no shader support)
00137         for (unsigned short t = 0, tCnt = ptrMat->getNumTechniques(); t < tCnt; ++t)
00138       {
00139             Technique *tech = ptrMat->getTechnique(t);
00140             for (unsigned short p = 0, pCnt = tech->getNumPasses(); p < pCnt; ++p)
00141          {
00142                 Pass *pass = tech->getPass(p);
00143                 //if (pass->getVertexProgramName() == "")
00144                 //  pass->setSpecular(0, 0, 0, 1);
00145             if (!pass->hasVertexProgram())
00146                pass->setSpecular(0.f, 0.f, 0.f, 1.f);
00147             }
00148         }
00149 
00150         //Store the original materials
00151         m_vecUnfadedMaterials.push_back(subBatch->getMaterial());

00178 {
00179     if (!m_bShadersSupported)

00189          m_pBatchGeom->setRenderQueueGroup(m_pPagedGeom ? m_pPagedGeom->getRenderQueue() : RENDER_QUEUE_6);
00190       else
00191          //Opaque batches should render in the normal render queue
00192          m_pBatchGeom->setRenderQueueGroup(RENDER_QUEUE_MAIN);
00193 
00194         m_fVisibleDist    = visibleDist;
