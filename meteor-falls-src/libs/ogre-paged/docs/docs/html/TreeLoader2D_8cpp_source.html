00026 TreeLoader2D::TreeLoader2D(PagedGeometry *geom, const TBounds &bounds)
00027 {
00028     //Calculate grid size
00029     TreeLoader2D::geom = geom;
00030     pageSize = geom->getPageSize();
00031 
00032     //Reset height function
00033     heightFunction = NULL;
00034     heightFunctionUserData = NULL;
00035 
00036     //Make sure the bounds are aligned with PagedGeometry's grid, so the TreeLoader's grid tiles will have a 1:1 relationship
00037     actualBounds = bounds;
00038     gridBounds = bounds;
00039     gridBounds.left = pageSize * Math::Floor((gridBounds.left - geom->getBounds().left) / pageSize) + geom->getBounds().left;
00040     gridBounds.top = pageSize * Math::Floor((gridBounds.top - geom->getBounds().top) / pageSize) + geom->getBounds().top;
00041     gridBounds.right = pageSize * Math::Ceil((gridBounds.right - geom->getBounds().left) / pageSize) + geom->getBounds().left;
00042     gridBounds.bottom = pageSize * Math::Ceil((gridBounds.bottom - geom->getBounds().top) / pageSize) + geom->getBounds().top;
00043 
00044     //Calculate page grid size
00045     pageGridX = (int)Math::Ceil(gridBounds.width() / pageSize) + 1;
00046     pageGridZ = (int)Math::Ceil(gridBounds.height() / pageSize) + 1;
00047 
00048     //Reset color map
00049     colorMap = NULL;
00050     colorMapFilter = MAPFILTER_NONE;
00051 
00052     //Default scale range
00053     maximumScale = 2.0f;
00054     minimumScale = 0.0f;

00104    else

00106         // If it does not exist, create a new page grid
00107         pageGrid = new std::vector<TreeDef>[pageGridX * pageGridZ];

00117     int pageX = (int)Math::Floor(xrel / pageSize); // bad perfomance float --> int
00118     int pageZ = (int)Math::Floor(zrel / pageSize); // bad perfomance float --> int

00120 
00121     //Create the new tree
00122     TreeDef tree;
00123     tree.xPos      = static_cast<uint16>(0xFFFF * (xrel - (pageX * pageSize)) / pageSize);
00124     tree.zPos      = static_cast<uint16>(0xFFFF * (zrel - (pageZ * pageSize)) / pageSize);

00132     //Add it to the tree list
00133     treeList.push_back(tree);

00441 void TreeLoader2D::loadPage(PageInfo &page)
00442 {
00443     //Calculate x/z indexes for the grid array
00444     page.xIndex -= (int)Math::Floor(gridBounds.left / pageSize);
00445     page.zIndex -= (int)Math::Floor(gridBounds.top / pageSize);
00446 
00447     //Check if the requested page is in bounds
00448     if (page.xIndex < 0 || page.zIndex < 0 || page.xIndex >= pageGridX || page.zIndex >= pageGridZ)
00449         return;
00450 
00451     //For each tree type...
00452     PageGridListIterator i;
00453     for (i = pageGridList.begin(); i != pageGridList.end(); ++i){
00454         //Get the appropriate tree list for the specified page
00455         std::vector<TreeDef> *pageGrid = i->second;
00456         std::vector<TreeDef> &treeList = _getGridPage(pageGrid, page.xIndex, page.zIndex);
00457         Entity *entity = i->first;
00458 
00459         //Load the listed trees into the page
00460         std::vector<TreeDef>::iterator o;
00461         for (o = treeList.begin(); o != treeList.end(); ++o){
00462             //Get position
00463             Vector3 pos;
00464             pos.x = page.bounds.left + ((Real)o->xPos / 65535) * pageSize;
00465             pos.z = page.bounds.top + ((Real)o->zPos / 65535) * pageSize;
00466 
00467             //Calculate terrain height at pos.x / pos.z to get pos.y
00468             if (heightFunction != NULL)
00469                 pos.y = heightFunction(pos.x, pos.z, heightFunctionUserData);
00470             else
00471                 pos.y = 0.0f;
00472             
00473             //Get rotation
00474             Degree angle((Real)o->rotation * (360.0f / 255));
00475             Quaternion rot(angle, Vector3::UNIT_Y);
00476 
00477             //Get scale
00478             Vector3 scale;
00479             scale.y = (Real)o->scale * (maximumScale / 255) + minimumScale;
00480             scale.x = scale.y;
00481             scale.z = scale.y;
00482 
00483             //Get color
00484             ColourValue color;
00485             if (colorMap)
00486                 color = colorMap->getColorAt_Unpacked(pos.x, pos.z, actualBounds);
00487             else
00488                 color = ColourValue::White;

00503 {

00506     //Test if the GridList has anything in it

00515     currentX = 0; currentZ = 0;

00517     currentTree = currentTreeList->begin();

00523 
00524     //Read the first tree's data

00526

00529     //it's too late)

00541 void TreeIterator2D::moveNext()
00542 {
00543     //Out of bounds check
00544     if (!hasMore)
00545         OGRE_EXCEPT(1, "Cannot read past end of TreeIterator list", "TreeIterator::moveNext()");
00546 
00547     //Preserve the last tree
00548     prevTreeDat = currentTreeDat;
00549 
00550     //Increment iterators to the next tree
00551     if (currentTree != currentTreeList->end())
00552         ++currentTree;
00553     while (currentTree == currentTreeList->end()){
00554         if (++currentX >= trees->pageGridX){
00555             currentX = 0;
00556             if (++currentZ >= trees->pageGridZ){
00557                 ++currentGrid;
00558                 if (currentGrid == trees->pageGridList.end()){
00559                     //No more trees left
00560                     hasMore = false;
00561                     return;
00562                 }
00563                 currentX = 0; currentZ = 0;
00564             }
00565         }
00566 
00567         currentTreeList = &trees->_getGridPage(currentGrid->second, currentX, currentZ);
00568         currentTree = currentTreeList->begin();
00569     }
00570 
00571     //Read the current tree data
00572     _readTree();
